# Operators
### In this section, we will learn about operators

```go
package main

import "fmt"

func main() {
    x := 10
    y := 5
    
    // Addition
    fmt.Println(x+y)
    
    // Subtraction
    fmt.Println(x-y)
    
    // Multiplication
    fmt.Println(x*y)
    
    // Division
    fmt. Println(x/y)
    
    // Modulus: result in remainder of the division
    fmt. Println(x%y)
    
    //The process of adding together strings is called string concatenation.
    p := "vikramkumar"
    q := "coder"
    fmt.Println(p + q)

}
```
